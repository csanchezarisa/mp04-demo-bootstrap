Barra de progrés

Les barres de progrés serveixen per informar a l'usuari de la quantitat de progrés que porta en un process.

Any - Primari
Mes - Èxit

Dia - Info
Hora - Advertència

Minut - Perill
Segon - Fosc

JavaScript

function updateBars() {
    const now = new Date();

    updateYear(now);
    updateDay(now);
    updateMonth(now);
    updateHour(now);
    updateMinute(now);
    updateSecond(now);

    window.requestAnimationFrame(updateBars);
}

function updateYear(now) {
    let start = new Date(now.getFullYear(), 0, 0);
    let diff = (now - start) + ((start.getTimezoneOffset() - now.getTimezoneOffset()) * 60 * 1000);
    let oneDay = 1000 * 60 * 60 * 24;
    let day = Math.floor(diff / oneDay);

    document.getElementById("bar-year").style.width = (day*100/365) + "%";
    document.getElementById("bar-year").innerHTML = Math.floor (day*100/365) + "%";
}

function updateMonth(now) {
    document.getElementById("bar-month").style.width = (now.getDate()*100/31) + "%";
    document.getElementById("bar-month").innerHTML= Math.floor (now.getDate()*100/31) + "%";
}

function updateDay(now) {
    document.getElementById("bar-day").style.width = (now.getHours()*100/24) + "%";
    document.getElementById("bar-day").innerHTML= Math.floor(now.getHours()*100/24) + "%";
}

function updateHour(now) {
    document.getElementById("bar-hour").style.width = (now.getMinutes()*100/60) + "%";
    document.getElementById("bar-hour").innerHTML = Math.floor (now.getMinutes()*100/60) + "%";
}

function updateMinute(now) {
    document.getElementById("bar-minute").style.width = (now.getSeconds()*100/60) + "%";
    document.getElementById("bar-minute").innerHTML = Math.floor (now.getSeconds()*100/60) + "%";
}

function updateSecond(now) {
    let ms = now.getMilliseconds();
    if(ms % 10 == 0) {
        document.getElementById("bar-second").style.width = Math.floor((ms*100/1000)) + "%";
        document.getElementById("bar-second").innerHTML = Math.floor((ms*100/1000)) + "%";
    }
}

window.onload = function() {
    window.requestAnimationFrame(updateBars);
}
